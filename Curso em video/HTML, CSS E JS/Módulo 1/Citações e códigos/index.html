 def obter_pagina(url):
        try:
            # Fazendo a requisição GET para a URL
            response = requests.get(url)
            
            # Verificando se a requisição foi bem-sucedida (código de status 200)
            if response.status_code == 200:
                # Retornando o conteúdo da página
                return response.text
            else:
                print(f"Erro ao obter página: {response.status_code}")
                return None
        except Exception as e:
            print(f"Ocorreu um erro: {str(e)}")
            return None
